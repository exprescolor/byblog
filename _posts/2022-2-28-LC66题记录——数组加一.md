## 解题思路:
> 分三种情况即可解决问题————数组中不含9 || 数组中含好几个9 || 数组中全为9
> 这道题比较重要的点在于:

代码如下：
  #include <iostream>
  #include <vector>
  using namespace std;

  vector<int> plusOne(vector<int> &digits)
  {
    int n = digits.size();
    for(int i  = n - 1; i > 0; i--)
    {
        if(digits[i] != 9)
        {
            digits[i]++;
            for(int j = i + 1; j < n; j++)
            {
              digits[j] = 0;
            }
            return digits;
        }
    }
    vector<int> ans(n + 1);
    ans[0] = 1;
    return ans;
  }
  int main() {
    vector<int> nums1;
    nums1.push_back(5);
    nums1.push_back(3);
    nums1.push_back(2);
    nums1.push_back(6);
    plusOne(nums1);

    cout << "hello..." << endl;
    return 0;
  }
