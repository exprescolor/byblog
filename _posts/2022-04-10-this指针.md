# this指针——指向成员函数所作用的对象

```CPP
class Complex
{
public:
	double real, imag;
	void Print()
	{
		cout << real << "." << imag;
	}
	Complex(double r, double i) : real(r), imag(i)
	{
		
	}
	Complex AddOne()
	{
		this->real++;		//等价于real++
		this->Print();		//等价于print
		return *this;
	}
};

int main()
{
	Complex c1(1, 1), c2(0, 0);
	c2 = c1.AddOne();           //c1的AddOne()调用时，this指针指向c1的real，因此c1.AddOne()调用后，实部为2虚部为1，统统赋值给c2
	return 0;
}
//程序输出 2.1，因为调用了一次Print()函数。
```
